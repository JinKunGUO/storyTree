```javascript:1536:1606:web/chapter.html
async function loadBreadcrumbNav(nodeId) {
    const response = await fetch(`/api/stories/nodes/${nodeId}/path`);
    const data = await response.json();
    
    // 渲染面包屑
    let breadcrumbHTML = '';
    
    // 添加故事标题
    if (data.story) {
        breadcrumbHTML += `
            <a href="/story?id=${data.story.id}" class="breadcrumb-item">
                <i class="fas fa-book"></i>
                <span>${data.story.title}</span>
            </a>
        `;
    }
    
    // 添加路径节点
    data.path.forEach((node, index) => {
        if (index === data.path.length - 1) {
            // 当前章节
            breadcrumbHTML += `
                <span class="breadcrumb-item current">
                    <i class="fas fa-file-alt"></i>
                    <span>${node.title}</span>
                </span>
            `;
        } else {
            // 可点击的历史节点
            breadcrumbHTML += `
                <a href="/chapter?id=${node.id}" class="breadcrumb-item">
                    <i class="fas fa-file-alt"></i>
                    <span>${node.title}</span>
                </a>
            `;
        }
    });
}
```

#### 后端 API
```typescript
// api/src/routes/nodes.ts
router.get('/:id/siblings', async (req, res) => {
  const { id } = req.params;
  
  const node = await prisma.nodes.findUnique({
    where: { id: parseInt(id) }
  });
  
  if (!node.parent_id) {
    return res.json({ siblings: [] });
  }
  
  const siblings = await prisma.nodes.findMany({
    where: { 
      parent_id: node.parent_id,
      id: { not: parseInt(id) }
    },
    include: {
      author: { select: { id: true, username: true } },
      _count: { select: { comments: true, ratings: true } }
    },
    orderBy: [
      { rating_avg: 'desc' },
      { read_count: 'desc' }
    ]
  });
  
  res.json({ siblings });
});
```

### 2. 分支合并建议（AI 推荐） - 未实现

**需求描述：**

**缺失原因：**
- 没有分支质量评估算法
- 没有 AI 推荐 API
- 没有推荐结果展示界面

**建议实现方案：**

#### 评分算法
```typescript
// 综合评分 = 阅读量权重 * 阅读量 + 评分权重 * 平均评分 + 评论权重 * 评论数
function calculateBranchScore(node) {
  const readWeight = 0.4;
  const ratingWeight = 0.4;
  const commentWeight = 0.2;
  
  const readScore = Math.log(node.read_count + 1) * readWeight;
  const ratingScore = (node.rating_avg || 0) * ratingWeight;
  const commentScore = Math.log(node._count.comments + 1) * commentWeight;
  
  return readScore + ratingScore + commentScore;
}
```

#### AI 推荐 API
```typescript
// api/src/routes/ai.ts
router.post('/recommend-branch', authenticateToken, async (req, res) => {
  const { storyId, nodeId } = req.body;
  
  // 获取所有子分支
  const branches = await prisma.nodes.findMany({
    where: { parent_id: parseInt(nodeId) },
    include: {
      author: true,
      _count: { select: { comments: true, ratings: true } }
    }
  });
  
  // 计算每个分支的得分
  const scoredBranches = branches.map(branch => ({
    ...branch,
    score: calculateBranchScore(branch)
  })).sort((a, b) => b.score - a.score);
  
  // 返回推荐结果
  res.json({
    recommended: scoredBranches[0],
    alternatives: scoredBranches.slice(1, 3),
    reason: `基于阅读量、评分和互动数据，推荐此分支`
  });
});
```
